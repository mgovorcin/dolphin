    class Settings(BaseSettings):
    """Settings configurable based on environment variables."""

    # https://docs.pydantic.dev/usage/settings/#parsing-environment-variable-values

    # If one is available, enable processing on GPU
    gpu_enabled: bool = True
    # Index of the GPU to use for processing, optional. Defaults to the
    # first available CUDA device. Ignored if *gpu_enabled* is False.
    gpu_id: int = 0
    # Number of cpu cores to use for processing (if CPU). Default 16
    n_workers: int = PositiveInt(16)

    class Config:
        env_prefix = "dolphin_"  # e.g. DOLPHIN_N_WORKERS=4 for n_workers
        fields = {
            "gpu_enabled": {"env": ["gpu_enabled", "gpu"]},
        }
